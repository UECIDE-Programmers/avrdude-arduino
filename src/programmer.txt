name=avrdude-arduino
description=Avrdude for Arduino boards
family=avr

board.reset.type=dtr

programmer.method=script
programmer.script.0=__builtin_bullet::Resetting board...
programmer.script.1=__builtin_exec::programmer.reset.${board.reset.type}.script
programmer.script.2=__builtin_bullet::Uploading sketch...
programmer.script.3.freebsd=${programmer:root}/freebsd/avrdude::${if:${board.erase}=yes,,-D}::-V::-p${build.mcu}::-P${port}::-b${upload.speed}::-C${programmer:root}/freebsd/avrdude.conf::-c${upload.protocol}::-Uflash:w:${build.path}/${filename}.hex:i::${programmer.verbose.${prefs:programmers.avrdude.verbose}}
programmer.script.3.linux_arm=${programmer:root}/linux_pi/avrdude::${if:${board.erase}=yes,,-D}::-V::-p${build.mcu}::-P${port}::-b${upload.speed}::-C${programmer:root}/linux_pi/avrdude.conf::-c${upload.protocol}::-Uflash:w:${build.path}/${filename}.hex:i::${programmer.verbose.${prefs:programmers.avrdude.verbose}}
programmer.script.3.linux_amd64=${programmer:root}/linux64/avrdude::${if:${board.erase}=yes,,-D}::-V::-p${build.mcu}::-P${port}::-b${upload.speed}::-C${programmer:root}/linux64/avrdude.conf::-c${upload.protocol}::-Uflash:w:${build.path}/${filename}.hex:i::${programmer.verbose.${prefs:programmers.avrdude.verbose}}
programmer.script.3.linux_i386=${programmer:root}/linux32/avrdude::${if:${board.erase}=yes,,-D}::-V::-p${build.mcu}::-P${port}::-b${upload.speed}::-C${programmer:root}/linux32/avrdude.conf::-c${upload.protocol}::-Uflash:w:${build.path}/${filename}.hex:i::${programmer.verbose.${prefs:programmers.avrdude.verbose}}
programmer.script.3.macosx=${programmer:root}/macosx/avrdude::${if:${board.erase}=yes,,-D}::-V::-p${build.mcu}::-P${port}::-b${upload.speed}::-C${programmer:root}/macosx/avrdude.conf::-c${upload.protocol}::-Uflash:w:${build.path}/${filename}.hex:i::${programmer.verbose.${prefs:programmers.avrdude.verbose}}
programmer.script.3.windows=${programmer:root}/windows/avrdude::${if:${board.erase}=yes,,-D}::-V::-p${build.mcu}::-P${port}::-b${upload.speed}::-C${programmer:root}/windows/avrdude.conf::-c${upload.protocol}::-Uflash:w:${build.path}/${filename}.hex:i::${programmer.verbose.${prefs:programmers.avrdude.verbose}}
programmer.script.4=__builtin_bullet::All done.

programmer.reset.baud.script.0=__builtin_port::open::1200
programmer.reset.baud.script.1=__builtin_delay::100
programmer.reset.baud.script.2=__builtin_port::close
programmer.reset.baud.script.3=__builtin_delay::1500

programmer.reset.dtr.script.0=__builtin_port::open::${upload.speed}
programmer.reset.dtr.script.1=__builtin_port::pulse
programmer.reset.dtr.script.2=__builtin_port::close


prefs.programmers.type=section
prefs.programmers.name=Programmers
prefs.programmers.avrdude.type=section
prefs.programmers.avrdude.name=AVRDude

prefs.programmers.avrdude.verbose.name=Verbosity
prefs.programmers.avrdude.verbose.type=dropdown
prefs.programmers.avrdude.verbose.options.none=None
prefs.programmers.avrdude.verbose.options.minimal=Minimal
prefs.programmers.avrdude.verbose.options.normal=Normal
prefs.programmers.avrdude.verbose.options.lots=Lots
prefs.programmers.avrdude.verbose.options.huge=Huge amounts
prefs.programmers.avrdude.verbose.options.insane=Insane
prefs.programmers.avrdude.verbose.default=none

# These map the option selected above to a set of flags through variable expansion
programmer.verbose.none=-q::-q
programmer.verbose.minimal=-q
programmer.verbose.normal=
programmer.verbose.lots=-v
programmer.verbose.huge=-v::-v
programmer.verbose.insane=-v::-v::-v::-v
